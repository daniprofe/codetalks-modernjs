Javascript or ES?

[A Brief History of Javascript](https://auth0.com/blog/a-brief-history-of-javascript/) Brendan Eich developed Javascript in only 10 days for Netscape. It was first released on Netscape Navigator 2.0, released on May 1995 It was firstly named 'Mocha', soon changed to 'LiveScript' and finally to 'Javascript' [ECMA](http://www.ecma-international.org) means 'European association for standardizing information and communication systems' Netscape submitted Javascript to ECMA for standarization on november 1996. The standard was named ECMA-262, aka ECMAScript, and is managed by the TC-39 comitee The first ECMAScript (ES1) was released with Netscape Navigator 4 on June 1997 * [List of EMCASCript engines](https://en.wikipedia.org/wiki/List_of_ECMAScript_engines) * [JavaScript Versions: ES6 and Before](https://www.codecademy.com/articles/javascript-versions)

Default parameters

How do you check for optional agruments and apply default values?


// chunkSize is optional, default value: 10

function processData (data, chunkSize) {
    console.log('Processing data on chunks of ' + chunkSize +
        ' records...');
}

processData(['ab', 'cd']);

Do you use this? Some problem with it?


// chunkSize is optional, default value: 10

function processData (data, chunkSize) {
    chunkSize = chunkSize || 10;
    console.log('Processing data on chunks of ' + chunkSize +
        ' records...');
}

processData(['ab', 'cd']);

Or this...


// chunkSize is optional, default value: 10

function processData (data, chunkSize) {
    if (!chunkSize) {
        chunkSize = 10;
    }

    console.log('Processing data on chunks of ' + chunkSize +
        ' records...');
}

processData(['ab', 'cd']);

Be careful: 'falsy'

The following are evaluated as false on boolean-context (if, &&, ||...)

  • false
  • null
  • 0
  • NaN
  • ''
  • ""
  • document.all

With ES6 we can use default parameters!


// chunkSize is optional, default value: 10

function processData (data, chunkSize = 10) {
    console.log('Processing data on chunks of ' + chunkSize +
        ' records...');
}

processData(['ab', 'cd']);

Template literals

Use the backtick, Luke!

How do you usually manage multi-line text?


let multilineText = 'This is a pretty story\n\n' +
    'Once upon a time...\n\n' +
    'THE END';

console.log(multilineText);

With ES6's template literals... funnier!


let multilineText = `This is a pretty story

    Once upon a time...

    THE END`;

console.log(multilineText);

How do you usually show variables 'inside' a string?


const apples = 8;
const kid = 'Peter';

let message = 'This morning ' + kid + ' bought ' +
    apples + ' apples';

console.log(message);

With ES6's template literals... funnier!


message = `This morning ${kid} bought ${apples} apples`;

console.log(message);

Any valid Javascript expression is valid inside dollar+brackets!


const total = 25;
message = `If ${kid} paid ${total} dollars,
    an apple costs ${total/apples} dollars`;

console.log(message);

Have you ever used Math.max?

0 to n arguments supported!


let max = Math.max(-2, 2, 800, 7);

And if your numbers are stored on an array?


const numbers = [-2, 2, 800, 7];

let max = Math.max.apply(this, numbers);

console.log(max);

On ES6 you can use spread operator!


max = Math.max(...numbers);

console.log(max);

	finalArray = [...array1, ...array2, 77, 88];

Rest operator: used on function arguments

Store the 'rest' of the parameters on an array

Should always be the last parameter of function


function MultMax(multiplier, ...numbers) {
    return multiplier * numbers.reduce(
        function(sum, number) {
            return sum + number;
        }
    );
}

console.log(MultSum(3, 1, 2));
console.log(MultSum(3, 1, 2, 3));

Array destructuring


let dimensions = [1, 2, 3];
let [width, height, depth] = dimensions;

console.log(`Width: ${width}`);
console.log(`Height: ${height}`);
console.log(`Depth: ${depth}`);

You can also mix spread operator and destructuring!


dimensions = [11, 22, 33, 4, 5, 6, 7];
[width, height, depth, ...other] = dimensions;

console.log(`Width: ${width}`);
console.log(`Height: ${height}`);
console.log(`Depth: ${depth}`);

Object destructuring


let worker = {
    name: 'John',
    age: 42,
    country: 'Spain'
};

let {name, age, country} = worker;

console.log(`My name is ${name}, I'm ${age}
    years old and I live in ${country}`);

You can of course use variable names not matching object properties


worker = {
    name: 'John',
    age: 42,
    country: 'Spain'
};

let {name: firstname, country: living} = worker;

console.log(`My name is ${firstname} and I live in ${living}`);

Be careful!


worker = {
    name: 'John',
    age: 42,
    country: 'Spain'
};

let workerName = 'Mary';

console.log(`My name is ${workerName}`);

let {name: workerName} = worker;

console.log(`My name is ${workerName}`);

This doesn't work!

You can also use mix spread operator and destructuring on objects!


worker = {
    name: 'John',
    age: 42,
    country: 'Spain'
};

let {name, ...otherData} = worker;

console.log(`My name is ${name}`);
console.log('Interesting info about me:');
console.log(otherData);

You can also use default values when destructuring


worker = {
    name: 'John',
    age: 42,
};

let {name, age, country = 'unknown'} = worker;

console.log(`My name is ${name}, I'm ${age}
    years old and I live in ${country}`);

And you can do this on function parameters too


function drawBox({width, height, color = 'yellow'}) {
    console.log(`Drawing box, dimensions: ${width}x${height},
        color: ${color}`);
}

drawBox({width: 4, height: 2});
drawBox({width: 3, height: 5, color: 'blue'});
